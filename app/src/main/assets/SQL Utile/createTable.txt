id INT(11) NOT NULL PRIMARY KEY auto_increment,

id_user INT(11),

status_relatie varchar(99),

copii varchar(99),

calatorii varchar(99),


id_categorie_1 INT(11),

id_categorie_2 INT(11),

id_locatii varchar(99)

);
